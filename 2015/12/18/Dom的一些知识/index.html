currentStyle.属性名；

function getStyle(ele,haha){
    if(window.getComputedStyle){
        return window.getComputedStyle(ele,null)[haha];
    }else if(ele.currentStyle){
        return ele.currentStyle[haha];
    }
    return null;
}
